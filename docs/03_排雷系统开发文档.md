## 一、背景与目标

### 1.1 业务背景

《企业财务排雷手册》定义了一套完整的企业财务分析框架，包含三个层次：

| 层次 | 目标 | 否决权 |
|------|------|--------|
| **排雷** | 识别财务造假（C01-C10 组合信号） | 🔴 直接排除 |
| **排险** | 识别生存风险（偿债能力） | 🔴 直接排除 |
| **排优** | 评估长期价值（评分卡） | 🟡 <60分不适合长期持有 |

### 1.2 开发目标

将排雷手册中的 **60+ 个可计算财务指标** 实现为插件系统，通过演化模式逐步完善。

```python
# vi_risk_analyzer/calculators/calc_gross_profit.py
"""
毛利 = 营业收入 - 营业成本
"""
REQUIRED_FIELDS = ["revenue", "cost"]

def calculate(data, config):
    """
    计算毛利
    
    Args:
        data: dict[str, pd.Series] - 字段数据
        config: dict - 用户配置
        
    Returns:
        pd.Series - 毛利
    """
    revenue = data["revenue"]
    cost = data["cost"]
    return revenue - cost
```

### 4.2 扫描器基类

```python
# vi_risk_analyzer/scanners/base.py
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Optional
import pandas as pd

@dataclass
class ScanResult:
    """扫描结果"""
    code: str              # C01-C10
    name: str              # 扫描器名称
    triggered: bool        # 是否触发
    severity: str          # ✅/⚠️/🔴/🟡
    value: float           # 计算值
    threshold: str         # 触发阈值
    description: str       # 说明

class BaseScanner(ABC):
    """扫描器基类"""
    
    code: str = ""
    name: str = ""
    
    @abstractmethod
    def scan(self, data: dict, years: int = 10) -> ScanResult:
        """执行扫描"""
        pass
    
    @abstractmethod
    def get_threshold(self) -> str:
        """获取触发阈值说明"""
        pass
```

### 4.3 扫描器实现示例

```python
# vi_risk_analyzer/scanners/c01_cash_flow_vs_profit.py
"""
C01: 现金流与利润背离

累计经营现金流/累计净利润 < 0.7（10年）= 高风险
"""
from .base import BaseScanner, ScanResult

class C01CashFlowVsProfitScanner(BaseScanner):
    code = "C01"
    name = "现金流与利润背离"
    
    def scan(self, data: dict, years: int = 10) -> ScanResult:
        """
        计算累计经营现金流/累计净利润
        
        多年累计值更能反映真实经营状况，单一年份易被操纵
        """
        ocf = data.get("operating_cash_flow", pd.Series())
        net_profit = data.get("net_profit", pd.Series())
        
        if len(ocf) < 5 or len(net_profit) < 5:
            return ScanResult(
                code=self.code,
                name=self.name,
                triggered=False,
                severity="🟡",
                value=float('nan'),
                threshold=self.get_threshold(),
                description=f"数据不足 {min(len(ocf), len(net_profit))} 年，无法判定"
            )
        
        # 使用累计值
        total_ocf = ocf.sum()
        total_profit = net_profit.sum()
        
        if total_profit == 0:
            ratio = float('nan')
        else:
            ratio = total_ocf / total_profit
        
        triggered = ratio < 0.7
        
        if ratio >= 0.7:
            severity = "✅"
        elif ratio >= 0.5:
            severity = "⚠️"
        else:
            severity = "🔴"
        
        return ScanResult(
            code=self.code,
            name=self.name,
            triggered=triggered,
            severity=severity,
            value=round(ratio, 2),
            threshold=self.get_threshold(),
            description=f"累计{len(ocf)}年: OCF={total_ocf:.2f}, 净利润={total_profit:.2f}, 比率={ratio:.2f}"
        )
    
    def get_threshold(self) -> str:
        return "累计经营现金流/累计净利润 < 0.7（10年）"
```

```python
from dataclasses import dataclass
from typing import Literal
import pandas as pd

@dataclass
class WarningResult:
    """预警结果"""
    priority: Literal["P0", "P1", "P2"]
    field: str
    value: float
    severity: str              # 🔴/⚠️/✅
    threshold: str
    description: str
    consecutive_years: int     # 连续异常年数

class WarningSystem:
    """预警系统"""
    
    def __init__(self, thresholds: dict = None):
        self.thresholds = thresholds or DEFAULT_THRESHOLDS
    
    def check_all(self, data: dict) -> list[WarningResult]:
        """检查所有预警指标"""
        results = []
        results.extend(self._check_p0(data))
        results.extend(self._check_p1(data))
        results.extend(self._check_p2(data))
        return results
    
    def _check_p0(self, data: dict) -> list[WarningResult]:
        """检查 P0 指标"""
        results = []
        
        # 1. 经营现金流/净利润
        result = self._check_consecutive_ratio(
            data, 
            "ocf_net_profit_ratio",
            "operating_cash_flow",
            "net_profit",
            threshold=1.0,
            consecutive_warning=2,
            priority="P0"
        )
        if result:
            results.append(result)
        
        # 2. 利息收入率
        result = self._check_interest_rate(data)
        if result:
            results.append(result)
        
        # ... 其他 P0 指标
        
        return results
    
    def _check_consecutive_ratio(
        self, 
        data: dict,
        name: str,
        numerator_field: str,
        denominator_field: str,
        threshold: float,
        consecutive_warning: int,
        priority: str
    ) -> WarningResult | None:
        """检查连续异常的比率指标"""
        numerator = data.get(numerator_field, pd.Series())
        denominator = data.get(denominator_field, pd.Series())
        
        if len(numerator) < consecutive_warning:
            return None
        
        ratios = numerator / denominator.replace(0, float('nan'))
        
        # 计算连续异常年数
        consecutive = 0
        for r in reversed(ratios):
            if r < threshold:
                consecutive += 1
            else:
                break
        
        latest = ratios.iloc[-1] if len(ratios) > 0 else float('nan')
        
        if consecutive >= consecutive_warning:
            severity = "🔴"
        elif consecutive >= 1:
            severity = "⚠️"
        else:
            severity = "✅"
        
        return WarningResult(
            priority=priority,
            field=name,
            value=round(latest, 2),
            severity=severity,
            threshold=f"连续{consecutive_warning}年 < {threshold}",
            description=f"最新值={latest:.2f}，连续{consecutive}年异常",
            consecutive_years=consecutive
        )
```

```python
# vi_risk_analyzer/scorers/long_term_scorer.py
from dataclasses import dataclass
import pandas as pd

@dataclass
class ScoreResult:
    """评分结果"""
    dimension: str
    weight: float
    score: float          # 0-100
    weighted_score: float
    details: dict

class LongTermScorer:
    """长期价值评分器"""
    
    # 评分标准
    STANDARDS = {
        # 长期增长
        "revenue_cagr": {"excellent": 0.15, "good": 0.08},      # 优秀>15%, 良好>8%
        "profit_cagr": {"excellent": 0.15, "good": 0.08},
        "growth_consistency": {"excellent": 0.80, "good": 0.60}, # 优秀>80%, 良好>60%
        
        # 稳定性
        "roe_volatility": {"excellent": 0.20, "good": 0.40},    # 优秀<20%, 良好<40%
        "margin_volatility": {"excellent": 0.05, "good": 0.10},
        "cash_flow_volatility": {"excellent": 0.20, "good": 0.40},
        "capex_stability": {"excellent": 0.03, "good": 0.05},
        
        # 可理解性
        "core_business_ratio": {"excellent": 0.90, "good": 0.70},
        "customer_concentration": {"excellent": 0.30, "good": 0.50},
    }
    
    def calculate_revenue_cagr(self, data: dict, years: int = 10) -> ScoreResult:
        """计算营收CAGR得分"""
        revenue = data.get("revenue", pd.Series())
        
        if len(revenue) < years:
            return ScoreResult(
                dimension="revenue_cagr",
                weight=0.0625,  # 25% / 4
                score=0,
                weighted_score=0,
                details={"error": f"数据不足 {len(revenue)} 年"}
            )
        
        start = revenue.iloc[0]
        end = revenue.iloc[-1]
        
        if start <= 0 or end <= 0:
            cagr = 0
        else:
            cagr = (end / start) ** (1 / (years - 1)) - 1
        
        # 评分
        if cagr >= self.STANDARDS["revenue_cagr"]["excellent"]:
            score = 100
        elif cagr >= self.STANDARDS["revenue_cagr"]["good"]:
            score = 75
        elif cagr >= 0:
            score = 50
        else:
            score = 25
        
        return ScoreResult(
            dimension="revenue_cagr",
            weight=0.0625,
            score=score,
            weighted_score=score * 0.0625,
            details={
                "cagr": round(cagr, 4),
                "start": start,
                "end": end,
                "years": years
            }
        )
    
    def calculate_roe_volatility(self, data: dict, years: int = 10) -> ScoreResult:
        """计算ROE波动率得分"""
        roe = data.get("roe", pd.Series())
        
        if len(roe) < 3:
            return ScoreResult(
                dimension="roe_volatility",
                weight=0.0875,  # 35% / 4
                score=0,
                weighted_score=0,
                details={"error": f"数据不足 {len(roe)} 年"}
            )
        
        mean = roe.mean()
        std = roe.std()
        
        if mean == 0:
            volatility = float('inf')
        else:
            volatility = std / abs(mean)
        
        # 评分（波动率越低越好）
        if volatility <= self.STANDARDS["roe_volatility"]["excellent"]:
            score = 100
        elif volatility <= self.STANDARDS["roe_volatility"]["good"]:
            score = 75
        elif volatility <= 0.6:
            score = 50
        else:
            score = 25
        
        return ScoreResult(
            dimension="roe_volatility",
            weight=0.0875,
            score=score,
            weighted_score=score * 0.0875,
            details={
                "volatility": round(volatility, 4),
                "mean": round(mean, 4),
                "std": round(std, 4)
            }
        )
```
